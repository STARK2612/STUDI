COMMANDE SQL POUR LA CREATION DE MA BASE DE DONNEE:

CREATE DATABASE zooarcadia;

USE zooarcadia;

CREATE TABLE utilisateur
(
    username VARCHAR(50) PRIMARY KEY, 
    password VARCHAR(50),
    nom      VARCHAR(50),
    prenom   VARCHAR(50),
    role_id  INT(11)
);

CREATE TABLE role
(
    role_id INT AUTO_INCREMENT PRIMARY KEY, 
    label   VARCHAR(50)
);

CREATE TABLE rapport_veterinaire
(
    rapport_veterinaire_id INT AUTO_INCREMENT PRIMARY KEY,
    date                   DATE,
    detail                 VARCHAR(50)
);

CREATE TABLE animal
(
    animal_id INT AUTO_INCREMENT PRIMARY KEY, 
    prenom    VARCHAR(50),
    etat      VARCHAR(50)
);

CREATE TABLE race
(
    race_id INT AUTO_INCREMENT PRIMARY KEY, 
    label   VARCHAR(50)
);

CREATE TABLE habitat
(
    habitat_id          INT AUTO_INCREMENT PRIMARY KEY,
    nom                 VARCHAR(50),
    description         VARCHAR(50),
    commentaire_habitat VARCHAR(50)
);

CREATE TABLE image
(
    image_id 	INT AUTO_INCREMENT PRIMARY KEY, 
    image_data  BLOB
);

CREATE TABLE avis
(
    avis_id             INT AUTO_INCREMENT PRIMARY KEY,
    pseudo              VARCHAR(50),
    commentaire         VARCHAR(50),
    isVisible           BOOL
);

CREATE TABLE service
(
    service_id  INT AUTO_INCREMENT PRIMARY KEY, 
    nom         VARCHAR(50),
    description VARCHAR(50)
);
